容易得出结论：可选择的只有第一个染色点，且不妨将该点视作根结点。该点确定后，本题答案也就确定了为所有子树的大小之和。现在要求的是如何选择根结点，使得答案最大。

设 fif_if​i​​ 为以 iii 为根的子树对答案的贡献。当 iii 为树根时，答案为 fif_if​i​​。sizisiz_isiz​i​​ 为 111 为树根时，以 iii 为根的子树的大小。不妨一开始以 111 为第一个染色点，然后进行换根 dp，按 dfs 序选择新的根。设 iii 为 111 的儿子，则可知

fi=n+∑k∈son(1)\ifk+n−sizi+∑k∈son(i)fkf_i=n+\sum \limits_{k\in son(1) \backslash i} f_k+n-siz_i+\sum \limits_{k\in son(i)} f_k f​i​​=n+​k∈son(1)\i​∑​​f​k​​+n−siz​i​​+​k∈son(i)​∑​​f​k​​

又因为

∑k∈son(i)fk=fi−sizi\sum \limits_{k\in son(i)} f_k=f_i-siz_i ​k∈son(i)​∑​​f​k​​=f​i​​−siz​i​​

和

∑k∈son(1)fk=f1−n\sum \limits_{k\in son(1)} f_k=f_1-n ​k∈son(1)​∑​​f​k​​=f​1​​−n

联立三式得到

fi=n+f1−2sizif_i=n+f_1-2siz_i f​i​​=n+f​1​​−2siz​i​​

这个例子是把根从 111 换到子结点 iii。同理，所有的 fif_if​i​​ 均可按照同样的方法由父结点的 fff 得到。最大的 fif_if​i​​ 为所求。

Code

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll N = 2e5 + 10;

ll n, f[N], ans, siz[N];
vector<ll> g[N];

void dfs(ll u, ll fa)
{
    siz[u] = 1;
    for (auto v : g[u]) {
        if (v == fa) continue;
        dfs(v, u);
        siz[u] += siz[v];
        f[u] += f[v];
    }
    f[u] += siz[u];
}

void dp(ll u, ll fa)
{
    ans = max(ans, f[u]);
    for (int v : g[u]) {
        if (v == fa) continue;
        f[v] = n + f[u] - 2 * siz[v];
        dp(v, u);
    }
}

int main()
{
    scanf("%lld", &n);
    for (int i = 1; i < n; i++) {
        ll u, v;
        scanf("%lld%lld", &u, &v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    dfs(1, 0); ans = f[1]; dp(1, 0);
    printf("%lld", ans);
    return 0;
}
