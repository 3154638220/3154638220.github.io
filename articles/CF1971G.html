int T, n, a[N], ans[N];

void WORK()
{
	map<int, vector<int> > mp;
	read(n);

		read(a[i]);

		mp[a[i] >> 2].pop_back();
	}

	putchar('\n');
}

int main()
{
	read(T);
	while (T--) WORK();
	return 0;
