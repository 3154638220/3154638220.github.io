#include <bits/stdc++.h>

D 优秀的拆分

Solution

本质上是求 nnn 的二进制转换，用位运算操作一下即可。

注意特判 nnn 为奇数时答案为 −1-1−1。

Code

#include <bits/stdc++.h>
using namespace std;

int n, x = (1 << 29);

int main()
{
    cin >> n;
    if (n & 1) 
        cout << "-1";
    else {
        while (x) {
            if (n & x)
                cout << x << ' ';
            x >>= 1;
        }
    }
    return 0;
}

F 植树节

Solution

差分模板题。定义 xix_ix​i​​ 的差分数组为 yiy_iy​i​​，满足 yi=xi−xi−1y_i=x_i-x_{i-1}y​i​​=x​i​​−x​i−1​​。显然 xix_ix​i​​ 和 yiy_iy​i​​ 是相互一一确定的。维护 yiy_iy​i​​ 就等于维护 xix_ix​i​​。

对 xxx 的 [l,r][l,r][l,r] 区间内的每个元素加一，在 yiy_iy​i​​ 中相当于 yl=yl+1y_{l}=y_{l}+1y​l​​=y​l​​+1 和 yr+1=yr+1−1y_{r+1}=y_{r+1}-1y​r+1​​=y​r+1​​−1。将原本在 xix_ix​i​​ 上 O(r−l+1)O(r-l+1)O(r−l+1) 的操作放在 yiy_iy​i​​ 上，就变成了 O(1)O(1)O(1) 的操作。

最后由 xi=xi−1+yix_i=x_{i-1}+y_ix​i​​=x​i−1​​+y​i​​，O(n)O(n)O(n) 求出 xix_ix​i​​ 即可。

Code

#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 10;

int n, x[N], ans;

int main()
{
    cin >> n;
    while (n--) {
        int l, r;
        cin >> l >> r;
        x[l]++;
        x[r + 1]--;
    }
    ans = x[0];
    for (int i = 1; i <= 1e6; i++) {
        x[i] += x[i - 1];
        ans = max(ans, x[i]);
    }
    cout << ans << endl;
    return 0;
}

F 多边形

Solution

注意到数据范围 n≤5000n\le 5000n≤5000，考虑 O(n2)O(n^2)O(n​2​​) 的 dp。

对木棍长度排序，然后枚举木棍 iii。

设 fxf_xf​x​​ 表示 长度和为 xxx 的木棍集合有多少种。将 x<lix<l_ix<l​i​​ 的所有 fxf_xf​x​​ 相加，即为以木棍 iii 为最长木棍时，多边形的方案数。

转移方程为

fx+li=fx+li+fxf_{x+l_i} = f_{x+l_i} + f_{x} f​x+l​i​​​​=f​x+l​i​​​​+f​x​​

Code

#include <bits/stdc++.h>
using namespace std;

const int mod = 998244353;

int n, maxn, ans;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    sort(a.begin(), a.end());
    maxn = a[n];
    vector<int> f(maxn + 10, 0);
    f[0] = 1;
    for (int i = 1; i <= n; i++) {
        if (i >= 3)
            for (int j = a[i] + 1; j <= maxn + 1; j++)
                ans = (ans + f[j]) % mod;
        for (int j = maxn + 1; j >= 0; j--) {
            if (!f[j]) continue;
            if (j + a[i] <= maxn)
                f[j + a[i]] = (f[j + a[i]] + f[j]) % mod;
            else
                f[maxn + 1] = (f[maxn + 1] + f[j]) % mod;
        }
    }
    cout << ans << endl;
    return 0;
}
