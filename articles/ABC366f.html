Solution

设 f1(x)=a1x+b1f_1(x)=a_1x+b_1f​1​​(x)=a​1​​x+b​1​​，f2(x)=a2x+b2f_2(x)=a_2x+b_2f​2​​(x)=a​2​​x+b​2​​。若 f1(f2(x))<f2(f1(x))f_1(f_2(x))<f_2(f_1(x))f​1​​(f​2​​(x))<f​2​​(f​1​​(x))，则 a1b2+a1<a2b1+b2a_1b_2+a_1<a_2b_1+b_2a​1​​b​2​​+a​1​​<a​2​​b​1​​+b​2​​。又因为 b1,b2>0b_1,b_2>0b​1​​,b​2​​>0，移项得a1−1b1<a2−1b2\dfrac{a_1-1}{b_1}<\dfrac{a_2-1}{b_2}​b​1​​​​a​1​​−1​​<​b​2​​​​a​2​​−1​​。

因此对于选定的 KKK 个 f(x)f(x)f(x)，从内到外 a−1b\dfrac{a-1}{b}​b​​a−1​​ 应升序排列。

我们对 NNN 个函数以 a−1b\dfrac{a-1}{b}​b​​a−1​​ 为关键字升序排序，用 01 背包求解即可。

Code

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll N = 2e5 + 10;

ll n, k, f[N];
struct Node {
	ll a, b;
	friend bool operator < (const Node &a, const Node &b) {
		return (a.a - 1) * b.b < (b.a - 1) * a.b;
	}
} a[N];

int main()
{
	scanf("%lld%lld", &n, &k);
	for (int i = 1; i <= n; i++)
		scanf("%lld%lld", &a[i].a, &a[i].b);
	sort(a + 1, a + 1 + n);
	f[0] = 1;
	for (int i = 1; i <= n; i++)
		for (int j = k; j; j--)
			f[j] = max(f[j], a[i].a * f[j - 1] + a[i].b);
	printf("%lld", f[k]);
	return 0;
}
