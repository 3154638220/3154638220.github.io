#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll N = 2e5 + 10;

ll n, m, a[N << 1], sum[1000010], ans;

int main()
{
    scanf("%lld%lld", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%lld", &a[i]);
        a[i + n] = a[i];
    }
    sum[0]++;
    for (int i = 1; i < n * 2; i++) {
        a[i] = (a[i] +  a[i - 1]) % m;
        sum[a[i]]++;
        if (i - n >= 0)
            sum[a[i - n]]--;
        ans += sum[a[i]] - 1;
        if (i >= n) 
            sum[a[i]]--; // йҳІз®—йҮҚ
    }
    printf("%lld", ans);
    return 0;
}
