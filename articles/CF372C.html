#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll N = 150010;

ll n, m, d, ans, f[2][N], minn = 1e16;
deque<ll> que;
struct Node {
    ll a, b, t;
} a[N];

int main()
{
    scanf("%lld%lld%lld", &n, &m, &d);
    for (int i = 1; i <= m; i++) {
        scanf("%lld%lld%lld", &a[i].a, &a[i].b, &a[i].t);
        ans += a[i].b;
    }
    for (int i = 1; i <= n; i++)
        f[1][i] = abs(a[1].a - i);
    for (int i = 2; i <= m; i++) {
        while (que.size()) que.pop_back();
        ll len = d * (a[i].t - a[i - 1].t);
        for (int j = 1; j <= n; j++)
            f[i & 1][j] = 1e16;
        for (int j = 1; j <= n; j++) {
            while (que.size() && que.front() + len < j)
                que.pop_front();
            while (que.size() && f[(i & 1) ^ 1][que.back()] >= f[(i & 1) ^ 1][j])
                que.pop_back();
            que.push_back(j);
            f[i & 1][j] = min(f[i & 1][j], f[(i & 1) ^ 1][que.front()] + abs(a[i].a - j));
        }
        while (que.size()) que.pop_back();
        for (int j = n; j; j--) {
            while (que.size() && que.front() - len > j)
                que.pop_front();
            while (que.size() && f[(i & 1) ^ 1][que.back()] >= f[(i & 1) ^ 1][j])
                que.pop_back();
            que.push_back(j);
            f[i & 1][j] = min(f[i & 1][j], f[(i & 1) ^ 1][que.front()] + abs(a[i].a - j));
        }
    }
    for (int i = 1; i <= n; i++)
        minn = min(minn, f[m & 1][i]);
    ans -= minn;
    printf("%lld", ans);
    return 0;
}
