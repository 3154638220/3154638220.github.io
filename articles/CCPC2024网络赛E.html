Code

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll mod = 998244353;

ll n, m, sum, maxn;

ll fst(ll a, ll b)
{
    ll res = 1;
    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

void exgcd(ll a, ll b, ll &x, ll &y)
{
    if (b == 0) {
        x = 1;
        y = 0;
        return;
    }
    exgcd(b, a % b, y, x);
    y = y - a / b * x;
}

int main()
{
    maxn = sum = 1;
    scanf("%lld%lld", &n, &m);
    ll mul = 1;
    for (int i = 1; i <= m; i++) {
        if (mul <= n) mul *= 26;
        maxn = (maxn + min(mul, n) + mod) % mod;
    }
    for (int i = 1; i <= m; i++) {
        ll inv, y, mi = fst(26, i);
        exgcd(mi, mod, inv, y);
        inv = (inv % mod + mod) % mod;
        sum = (sum + (1 - fst(1 - inv, n) + mod) % mod * fst(26, i) % mod) % mod;
    }
    printf("%lld %lld", maxn, sum);
    return 0;
}
